06 Core Implementation

⚠️
Key SDK Difference vs OpenAI: The Sarvam SDK does not use .create(). Call .completions() directly — no .create() suffix.

Chat with Conversation Memory

python — chat.py
conversation_history = []
MAX_HISTORY = 20  # last 20 messages = 10 turns

system_prompt = {
    "role": "system",
    "content": """You are a helpful AI assistant. Respond in
the language the user writes in. Keep replies concise."""
}

def chat(user_message: str) -> str:
    conversation_history.append({"role": "user", "content": user_message})

    response = client.chat.completions(
        model="sarvam-105b",
        messages=[system_prompt] + conversation_history[-MAX_HISTORY:]
    )

    reply = response.choices[0].message.content
    conversation_history.append({"role": "assistant", "content": reply})
    return reply

Language Detection

python — language.py
def detect_language(text: str) -> str:
    """Returns BCP-47 code: hi-IN, ta-IN, ml-IN, etc."""
    response = client.text.identify_language(input=text)
    return response.language_code

LANGUAGE_NAMES = {
    "hi-IN": "Hindi",   "ta-IN": "Tamil",   "te-IN": "Telugu",
    "ml-IN": "Malayalam", "kn-IN": "Kannada",  "bn-IN": "Bengali",
    "mr-IN": "Marathi",  "gu-IN": "Gujarati", "pa-IN": "Punjabi",
    "od-IN": "Odia",    "as-IN": "Assamese", "ur-IN": "Urdu",
    "en-IN": "English"
}

Dynamic Language-Aware Chat

python
def chat(user_message: str) -> tuple[str, str]:
    lang_code = detect_language(user_message)
    lang_name = LANGUAGE_NAMES.get(lang_code, "Hindi")

    conversation_history.append({"role": "user", "content": user_message})

    dynamic_system = {
        "role": "system",
        "content": f"You are a helpful assistant. The user writes in {lang_name}. "
                   f"Always respond in {lang_name}. Be warm and concise."
    }

    response = client.chat.completions(
        model="sarvam-105b",
        messages=[dynamic_system] + conversation_history[-MAX_HISTORY:]
    )

    reply = response.choices[0].message.content
    conversation_history.append({"role": "assistant", "content": reply})
    return reply, lang_code

Translation

python — translation.py
def translate_text(text: str, target_language: str,
                    source_language: str = "auto") -> str:
    response = client.text.translate(
        input=text,
        source_language_code=source_language,
        target_language_code=target_language,
        model="sarvam-translate:v1"
    )
    return response.translated_text

# Formal style (mayura:v1)
client.text.translate(
    input="Your payment is pending",
    source_language_code="en-IN",
    target_language_code="hi-IN",
    model="mayura:v1",
    mode="formal"
)

# Colloquial style
client.text.translate(
    input="Your payment is pending",
    source_language_code="en-IN",
    target_language_code="hi-IN",
    model="mayura:v1",
    mode="modern-colloquial"
)

07 Streamlit UI

💡
Run the complete UI with: uv run streamlit run chatbot.py
python — chatbot.py (full)
import streamlit as st
import os
from dotenv import load_dotenv
from sarvamai import SarvamAI

load_dotenv()
client = SarvamAI(api_subscription_key=os.environ["SARVAM_API_KEY"])

st.set_page_config(page_title="Indian Language Chatbot", page_icon="🇮🇳")
st.title("🇮🇳 Indian Language Chatbot")
st.caption("Powered by Sarvam AI - 22 Indian languages")

# Sidebar settings
with st.sidebar:
    st.header("Settings")
    response_language = st.selectbox(
        "Response Language",
        ["Auto-detect", "Hindi", "Tamil", "Telugu",
         "Malayalam", "Kannada", "Bengali", "Marathi", "Gujarati"]
    )
    if st.button("Clear conversation"):
        st.session_state.messages = []
        st.rerun()

# Session state for chat history
if "messages" not in st.session_state:
    st.session_state.messages = []

# Render previous messages
for msg in st.session_state.messages:
    with st.chat_message(msg["role"]):
        st.markdown(msg["content"])

# Chat input
if prompt := st.chat_input("Type in any language..."):
    st.session_state.messages.append({"role": "user", "content": prompt})
    with st.chat_message("user"):
        st.markdown(prompt)

    with st.chat_message("assistant"):
        response = client.chat.completions(
            model="sarvam-105b",
            messages=st.session_state.messages[-20:]
        )
        reply = response.choices[0].message.content
        st.markdown(reply)

    st.session_state.messages.append({"role": "assistant", "content": reply})

03 Database Schema

For production deployment, a PostgreSQL relational database persists conversations, user sessions, language preferences, and usage analytics.

Entity Relationships

users
  ├── user_id (PK)
  ├── [1] ←────── [N] sessions.user_id (FK)
  └── [1] ←────── [1] user_preferences.user_id (FK, UNIQUE)

sessions
  ├── session_id (PK)
  ├── user_id (FK → users)
  ├── [1] ←────── [N] messages.session_id (FK)
  ├── [1] ←────── [N] language_detections.session_id (FK)
  └── [1] ←────── [N] translations.session_id (FK)

messages
  ├── message_id (PK)
  ├── session_id (FK → sessions)
  ├── [1] ←────── [N] language_detections.message_id (FK)
  └── [1] ←────── [N] translations.message_id (FK, nullable)

13 Security Checklist

  • NEVER commit .env to git — always add to .gitignore before first commit
  • NEVER hardcode SARVAM_API_KEY in source code
  • Use platform secret managers (Railway / Render / AWS SSM) in production
  • Hash all passwords with bcrypt before storing in DB
  • Use HTTPS only for any deployed endpoint
  • Rotate API keys every 90 days
  • Rate-limit the chat endpoint to prevent abuse
